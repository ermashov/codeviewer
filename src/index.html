      // Объявление функции для подсчета суммы
int вычеслить_сумму(int a, int b) {
    return a + b;
}

int main() {
    // Объявление переменных
    int число1, число2, результат;

    printf("Привет! Это программа на языке C.\n");

    // Запрос данных у пользователя
    printf("Введите первое целое число: ");
    scanf("%d", &число1);

    printf("Введите второе целое число: ");
    scanf("%d", &число2);

    // Вызов функции и сохранение результата
    результат = вычеслить_сумму(число1, число2);

    // Вывод итогового результата на экран
    printf("Результат: %d + %d = %d\n", число1, число2, результат);

    return 0; // Признак успешного завершения программы
}
