class Product

    public string Name { get; set; }
    public decimal Price { get; set; }

        Name = name;

    }

    static void Main()
    {
        Console.WriteLine("--- Список покупок на C# ---");

class Student {

;; Определение списка сотрудников (каждый сотрудник — это карта/map)
(def employees
  [{:name "Алексей" :department "IT"      :salary 120000}
   {:name "Мария"   :department "Design"  :salary 95000}
   {:name "Иван"    :department "IT"      :salary 140000}
   {:name "Елена"   :department "HR"      :salary 80000}])

;; Функция для фильтрации и расчета средней зарплаты в конкретном отделе
(defn average-salary-by-dept [emp-list dept-name]
  ;; 1. Фильтруем сотрудников по нужному отделу
  (let [filtered-emps (filter #(= (:department %) dept-name) emp-list)
        ;; 2. Извлекаем только зарплаты
        salaries      (map :salary filtered-emps)
        ;; 3. Считаем сумму всех зарплат
        total-salary  (reduce + salaries)
        ;; 4. Считаем количество сотрудников в отделе
        emp-count     (count filtered-emps)]

    ;; Проверяем, нашли ли мы сотрудников, чтобы избежать деления на ноль
    (if (> emp-count 0)
      (/ total-salary emp-count)
      0)))

;; Главная функция (точка входа)
(defn -main []
  (println "=== Анализ зарплат на Clojure ===")

  ;; Выводим результат работы функции для IT-отдела
  (let [it-avg (average-salary-by-dept employees "IT")]
    (println "Средняя зарплата в IT отделе:" (float it-avg) "руб."))

  ;; Пример работы с неизменяемостью: добавим сотрудника и выведем общее число
  ;; Исходный список employees при этом НЕ изменится
  (let [new-employees (conj employees {:name "Ольга" :department "IT" :salary 110000})]
    (println "Исходное кол-во сотрудников:" (count employees))
    (println "Новое кол-во в измененной копии:" (count new-employees))))

;; Запуск главной функции (для демонстрации в REPL)
(-main)

       // Определение класса "Задача" (Task)
class Task {
  final String title;
  bool isCompleted;
  // Именованный параметр в конструкторе с дефолтным значением
  Task({required this.title, this.isCompleted = false});

  // Метод для переключения статуса задачи
  void toggleStatus() {
    isCompleted = !isCompleted;
  }

  @override
  String toString() {
    return '[${isCompleted ? 'X' : ' '}] $title';
  }
}

// Главная функция — точка входа в программу
void main() {
  print('=== Менеджер задач на Dart ===');

  // Создание списка объектов (динамический массив)
  final taskList = [
    Task(title: 'Купить молоко'),
    Task(title: 'Написать код на Dart', isCompleted: true),
    Task(title: 'Сделать зарядку'),
  ];

  // Добавление нового элемента в список
  taskList.add(Task(title: 'Почитать книгу'));

  // Изменение статуса одной из задач
  taskList[0].toggleStatus();

  // Вывод всех задач с помощью метода forEach
  print('\nВсе задачи:');
  taskList.forEach((task) => print(task));

  // Фильтрация списка с помощью стрелочной функции
  final completedCount = taskList.where((task) => task.isCompleted).length;

  print('------------------------------');
  print('Выполнено задач: $completedCount из ${taskList.length}');
}

      program ArrayProcessing
    implicit none  ! Отключает неявное приведение типов (хорошая практика)

    ! Объявление переменных
    integer, parameter :: size = 5
    real, dimension(size) :: vector
    real :: average_value
    integer :: i

    print *, "=== Анализ данных на Fortran ==="

    ! Инициализация массива значениями
    do i = 1, size
        vector(i) = real(i) * 2.5
    end do

    ! Вывод элементов массива на экран
    print *, "Элементы массива:"
    do i = 1, size
        print '(A, I2, A, F5.1)', "Элемент (", i, ") = ", vector(i)
    end do

    ! Вызов пользовательской функции для расчета среднего значения
    average_value = calculate_average(vector, size)

    print *, "---------------------------------"
    print '(A, F5.2)', "Среднее значение элементов: ", average_value

contains

    ! Описание функции внутри программы
    function calculate_average(arr, n) result(avg)
        integer, intent(in) :: n
        real, dimension(n), intent(in) :: arr
        real :: avg
        real :: total_sum
        integer :: j

        total_sum = 0.0
        do j = 1, n
            total_sum = total_sum + arr(j)
        end do

        avg = total_sum / real(n)
    end function calculate_average

end program ArrayProcessing
        
      
        
      package main

import (
	"fmt"
)

// Определение структуры "Автомобиль"
type Car struct {
	Brand string
	Model string
	Year  int
}

// Метод для структуры Car (аналог метода класса в ООП)
func (c Car) Info() string {
	return fmt.Sprintf("%s %s (%d г.)", c.Brand, c.Model, c.Year)
}

func main() {
	fmt.Println("=== Автопарк на Go ===")

	// Создание среза (slice) — динамического массива из структур
	cars := []Car{
		{Brand: "Toyota", Model: "Camry", Year: 2021},
		{Brand: "Tesla", Model: "Model 3", Year: 2023},
		{Brand: "BMW", Model: "M5", Year: 2019},
	}

	// Добавление нового элемента в срез
	cars = append(cars, Car{Brand: "Lada", Model: "Vesta", Year: 2024})

	// Цикл для перебора элементов (range возвращает индекс и сам элемент)
	for i, car := range cars {
		// Используем метод Info() для вывода информации
		fmt.Printf("%d. %s\n", i+1, car.Info())
	}

	fmt.Println("----------------------")
	fmt.Printf("Всего автомобилей в базе: %d\n", len(cars))
}
